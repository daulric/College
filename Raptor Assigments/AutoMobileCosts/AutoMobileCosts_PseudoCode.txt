Declare double insurance;
Declare double loan_payment;
Declare double gas;
Declare double oil;

Declare double maintainace;

Declare double monthly_cost;
Declare double yearly_cost;

// Getting Inputs
Display "Enter Loan Payment";

Input insurance;

Input gas;

Input oil;
